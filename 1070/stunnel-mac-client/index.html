    stunnel Mac client

    stunnel server 的設定就照這篇的教學，不過 Mac client 的設定似乎不能直接套用 Ubuntu client 的，試了一下網路上的一些設定，總算有可以 work 的了，以下是我的設定，比較要注意的就是 cert 的位置要設對，另外就是 connect 的 host 要改成你 stunnel server 的位置：

    setuid = nobody
setgid = nogroup
pid = /tmp/stunnel.pid

cert = /usr/local/etc/stunnel/client.pem
client = yes

debug = 7

sslVersion = SSLv3
verify = 0
[danted]
accept  = 127.0.0.1:1080
connect = host:1081
